# Ceramic Tile Production Line Simulator - Supabase Implementation Guide

## 🎯 Objective

Migrate the ceramic tile production line simulator from Zustand local state to Supabase persistent database while maintaining real-time simulation capabilities. This will enable:
- Persistent storage of simulation data across sessions
- AI-powered root cause analysis of defects
- Historical comparison between simulation runs
- Multi-user access to simulation results

## 🗄️ Database Schema

Two files are provided:
1. **`ceramic_simulator_schema.sql`** - Complete PostgreSQL schema for Supabase
2. **`simulator-types.ts`** - TypeScript type definitions

### Key Tables Overview

| Table | Purpose | Write Frequency |
|-------|---------|-----------------|
| `simulation_sessions` | Session management | Once per simulation |
| `machine_*_states` | Machine snapshots per tick | Every sim tick (7 tables) |
| `tiles` | Tile lifecycle tracking | On tile creation/completion |
| `tile_station_snapshots` | Tile-machine associations (künye) | When tile passes station |
| `parameter_change_events` | Parameter change log | On random changes |
| `production_metrics` | Aggregated KPIs | Periodic (every N ticks) |

### Important: Session Code
Each simulation has a unique 6-character `session_code` (e.g., "A3F2B1") auto-generated by the database. Use this for:
- Displaying to users
- Filtering queries
- Referencing simulations in AI analysis

  const { data: session, error } = await supabase

export async function updateSimulationStatus(
  sessionId: string,
  status: SimulationSession['status'],
  tick?: { sim: number; production: number }
): Promise<void> {
  const updates: Partial<SimulationSession> = { status };
  
  if (tick) {
    updates.current_sim_tick = tick.sim;
    updates.current_production_tick = tick.production;
  }
  
  if (status === 'running' && !tick) {
    updates.started_at = new Date().toISOString();
  } else if (status === 'paused') {
    updates.paused_at = new Date().toISOString();
  } else if (status === 'completed' || status === 'aborted') {
    updates.completed_at = new Date().toISOString();
  }

  const { error } = await supabase
    .from('simulation_sessions')
    .update(updates)
    .eq('id', sessionId);

  if (error) throw error;
}

export async function getSimulationByCode(
  sessionCode: string
): Promise<SimulationSession | null> {
  const { data, error } = await supabase
    .from('simulation_sessions')
    .select()
    .eq('session_code', sessionCode.toUpperCase())
    .single();

  if (error && error.code !== 'PGRST116') throw error;
  return data;
}

// Batch insert for all machines at once (more efficient)
export async function recordAllMachineStates(
  simulationId: string,
  simTick: number,
  productionTick: number,
  states: {
    press: PressParameters;
    dryer: DryerParameters;
    glaze: GlazeParameters;
    printer: PrinterParameters;
    kiln: KilnParameters;
    sorting: SortingParameters;
    packaging: PackagingParameters;
  }
): Promise<Record<StationName, string>> {
  const results: Record<string, string> = {};

  // Use Promise.all for parallel inserts
  const inserts = Object.entries(states).map(async ([station, params]) => {
    const tableName = MACHINE_TABLES[station as StationName];
    
    const { data, error } = await supabase
      .from(tableName)
      .insert({
        simulation_id: simulationId,
        sim_tick: simTick,
        production_tick: productionTick,
        is_operating: true,
        ...params,
      })
      .select('id')
      .single();

    if (error) throw error;
    results[station] = data.id;
  });

  await Promise.all(inserts);
  return results as Record<StationName, string>;
}

    .from('parameter_change_events')
    .insert({
      simulation_id: simulationId,
      sim_tick: simTick,
      production_tick: productionTick,
      ...change,
      change_magnitude: changeMagnitude,
      change_pct: changePct,
    });

  if (error) throw error;
}

### Step 5: Create Write Buffer for Performance

Since simulation runs at 500ms ticks, we need buffered writes to avoid overwhelming Supabase.

Create `src/services/writeBuffer.ts`:

```typescript
import { supabase } from '@/lib/supabase';

interface BufferedWrite {
  table: string;
  data: Record<string, unknown>;
  timestamp: number;
}

class WriteBuffer {
  private buffer: BufferedWrite[] = [];
  private flushInterval: NodeJS.Timeout | null = null;
  private readonly FLUSH_INTERVAL_MS = 2000; // Flush every 2 seconds
  private readonly MAX_BUFFER_SIZE = 100;

  constructor() {
    this.startFlushInterval();
  }

  add(table: string, data: Record<string, unknown>) {
    this.buffer.push({ table, data, timestamp: Date.now() });
    
    if (this.buffer.length >= this.MAX_BUFFER_SIZE) {
      this.flush();
    }
  }

  private startFlushInterval() {
    this.flushInterval = setInterval(() => {
      if (this.buffer.length > 0) {
        this.flush();
      }
    }, this.FLUSH_INTERVAL_MS);
  }

  async flush() {
    if (this.buffer.length === 0) return;

    const toFlush = [...this.buffer];
    this.buffer = [];

    // Group by table for batch inserts
    const grouped = toFlush.reduce((acc, item) => {
      if (!acc[item.table]) acc[item.table] = [];
      acc[item.table].push(item.data);
      return acc;
    }, {} as Record<string, Record<string, unknown>[]>);

    // Execute batch inserts in parallel
    const promises = Object.entries(grouped).map(([table, records]) =>
      supabase.from(table).insert(records as any)
    );

    try {
      await Promise.all(promises);
      console.log(`Flushed ${toFlush.length} records to database`);
    } catch (error) {
      console.error('Buffer flush error:', error);
      // Re-add failed items to buffer for retry
      this.buffer.unshift(...toFlush);
    }
  }

  stop() {
    if (this.flushInterval) {
      clearInterval(this.flushInterval);
      this.flushInterval = null;
    }
    // Final flush
    return this.flush();
  }
}

export const writeBuffer = new WriteBuffer();
```

// =============================================================================
// STORE INTERFACE
// =============================================================================

interface SimulationStore {
  // ---------------------------------------------------------------------------
  // SESSION
  // ---------------------------------------------------------------------------
  session: SimulationSession | null;
  sessionCode: string | null;
  
  // ---------------------------------------------------------------------------
  // RUNTIME STATE
  // ---------------------------------------------------------------------------
  isRunning: boolean;
  currentSimTick: number;
  currentProductionTick: number;
  
  // ---------------------------------------------------------------------------
  // MACHINE STATE TABLES (Historical records, indexed by sim_tick)
  // ---------------------------------------------------------------------------
  machineStates: MachineStateTables;
  
  // ---------------------------------------------------------------------------
  // CURRENT MACHINE PARAMETERS (Live values)
  // ---------------------------------------------------------------------------
  currentParams: CurrentMachineParams;
  
  // ---------------------------------------------------------------------------
  // MACHINE OPERATING STATUS
  // ---------------------------------------------------------------------------
  machineStatus: Record<StationName, { isOperating: boolean; faultCode?: string }>;
  
  // ---------------------------------------------------------------------------
  // TILES
  // ---------------------------------------------------------------------------
  tiles: Map<string, TileRecord>;                    // All tiles, keyed by id
  tilesByNumber: Map<number, string>;                // tile_number -> tile_id
  
  // ---------------------------------------------------------------------------
  // TILE SNAPSHOTS (KÜNYE)
  // ---------------------------------------------------------------------------
  tileSnapshots: Map<string, TileSnapshotRecord[]>;  // tile_id -> snapshots[]
  
  // ---------------------------------------------------------------------------
  // CONVEYOR (Runtime positions)
  // ---------------------------------------------------------------------------
  conveyorPositions: Map<string, ConveyorPosition>;  // tile_id -> position
  
  // ---------------------------------------------------------------------------
  // PARAMETER CHANGES
  // ---------------------------------------------------------------------------
  parameterChanges: ParameterChangeRecord[];
  
  // ---------------------------------------------------------------------------
  // SCENARIO ACTIVATIONS
  // ---------------------------------------------------------------------------
  activeScenarios: Map<string, ScenarioActivationRecord>;  // scenario_id -> activation
  scenarioHistory: ScenarioActivationRecord[];
  
  // ---------------------------------------------------------------------------
  // PRODUCTION METRICS
  // ---------------------------------------------------------------------------
  metricsHistory: ProductionMetricsRecord[];
  currentPeriodMetrics: {
    periodStartTick: number;
    totalProduced: number;
    firstQuality: number;
    secondQuality: number;
    thirdQuality: number;
    scrap: number;
    scrapByStation: Record<StationName, number>;
    defectCounts: Record<string, number>;
  };
  
  // ---------------------------------------------------------------------------
  // COUNTERS & STATS
  // ---------------------------------------------------------------------------
  tileCounter: number;
  totalTilesProduced: number;
  totalTilesScrapped: number;
  
  // ---------------------------------------------------------------------------
  // SYNC TRACKING
  // ---------------------------------------------------------------------------
  unsyncedRecords: {
    machineStates: { station: StationName; simTick: number }[];
    tiles: string[];
    snapshots: string[];
    parameterChanges: string[];
    scenarios: string[];
    metrics: string[];
  };
  lastSyncTick: number;
  
  // ---------------------------------------------------------------------------
  // CONFIG
  // ---------------------------------------------------------------------------
  config: {
    tickDurationMs: number;
    productionTickRatio: number;
    stationGapProductionTicks: number;
    metricsPeriodTicks: number;
    syncBatchSize: number;
    parameterChangeChance: number;
  };
  
  // ---------------------------------------------------------------------------
  // SCENARIOS (Loaded from DB)
  // ---------------------------------------------------------------------------
  defectScenarios: DefectScenario[];
  
  // ---------------------------------------------------------------------------
  // ACTIONS
  // ---------------------------------------------------------------------------
  
  // Session management
  initializeSimulation: (name: string, description?: string) => Promise<void>;
  loadSimulation: (sessionCode: string) => Promise<void>;
  startSimulation: () => void;
  pauseSimulation: () => void;
  stopSimulation: () => Promise<void>;
  resetSimulation: () => void;
  
  // Main tick
  tick: () => void;
  
  // Machine state management
  recordMachineState: (station: StationName, simTick: number, productionTick: number) => string;
  updateMachineParam: <S extends StationName>(
    station: S,
    param: string,
    value: number,
    changeType?: 'drift' | 'spike' | 'random' | 'step'
  ) => void;
  setMachineFault: (station: StationName, faultCode: string) => void;
  clearMachineFault: (station: StationName) => void;
  
  // Tile management
  createTile: (simTick: number, productionTick: number) => TileRecord;
  updateTileStatus: (tileId: string, status: TileRecord['status']) => void;
  setTileGrade: (tileId: string, grade: QualityGrade) => void;
  scrapTile: (tileId: string, station: StationName, defects: DefectType[]) => void;
  
  // Snapshot management
  recordTileSnapshot: (
    tileId: string,
    station: StationName,
    simTick: number,
    productionTick: number,
    machineStateId: string,
    defectInfo?: { detected: boolean; types?: DefectType[]; severity?: number; scrapped?: boolean }
  ) => string;
  
  // Conveyor management
  moveTilesOnConveyor: (simTick: number, productionTick: number) => void;
  
  // Scenario management
  checkAndActivateScenarios: (simTick: number) => void;
  deactivateScenario: (scenarioId: string, simTick: number) => void;
  
  // Metrics
  updatePeriodMetrics: (simTick: number, productionTick: number) => void;
  finalizePeriodMetrics: (simTick: number, productionTick: number) => void;
  
  // Sync
  markForSync: (type: keyof SimulationStore['unsyncedRecords'], id: string | { station: StationName; simTick: number }) => void;
  getUnsyncedData: () => {
    machineStates: Record<StationName, any[]>;
    tiles: TileRecord[];
    snapshots: TileSnapshotRecord[];
    parameterChanges: ParameterChangeRecord[];
    scenarios: ScenarioActivationRecord[];
    metrics: ProductionMetricsRecord[];
  };
  markAsSynced: (type: keyof SimulationStore['unsyncedRecords'], ids: string[]) => void;
  
  // Queries
  getTileById: (id: string) => TileRecord | undefined;
  getTileByNumber: (number: number) => TileRecord | undefined;
  getTileSnapshots: (tileId: string) => TileSnapshotRecord[];
  getMachineStateAtTick: <S extends StationName>(station: S, simTick: number) => any;
  getRecentParameterChanges: (station?: StationName, limit?: number) => ParameterChangeRecord[];
}

    metricsHistory: [],

      scrap: 0,
      scrapByStation: {
        press: 0, dryer: 0, glaze: 0, printer: 0, kiln: 0, sorting: 0, packaging: 0,
      },
      defectCounts: {},
    },
    
    tileCounter: 0,
    totalTilesProduced: 0,
    totalTilesScrapped: 0,
    
    unsyncedRecords: {
      machineStates: [],
      tiles: [],
      snapshots: [],
      parameterChanges: [],
      scenarios: [],
      metrics: [],
    },
    lastSyncTick: 0,
    
    config: {
      tickDurationMs: 500,
      productionTickRatio: 2,
      stationGapProductionTicks: 2,
      metricsPeriodTicks: 100,
      syncBatchSize: 50,
      parameterChangeChance: 0.02,
    },
    
    defectScenarios: [],
    
    // -------------------------------------------------------------------------
    // SESSION MANAGEMENT
    // -------------------------------------------------------------------------
    initializeSimulation: async (name, description) => {
      // This will be implemented with Supabase integration
      // For now, create local session
      const sessionId = nanoid();
      const sessionCode = nanoid(6).toUpperCase();
      
      const session: SimulationSession = {
        id: sessionId,
        session_code: sessionCode,
        name,
        description,
        status: 'created',
        current_sim_tick: 0,
        current_production_tick: 0,
        tick_duration_ms: get().config.tickDurationMs,
        production_tick_ratio: get().config.productionTickRatio,
        station_gap_production_ticks: get().config.stationGapProductionTicks,
        created_at: new Date().toISOString(),
        updated_at: new Date().toISOString(),
      };
      
      set({
        session,
        sessionCode,
        currentSimTick: 0,
        currentProductionTick: 0,
        tiles: new Map(),
        tilesByNumber: new Map(),
        tileSnapshots: new Map(),
        conveyorPositions: new Map(),
        parameterChanges: [],
        activeScenarios: new Map(),
        scenarioHistory: [],
        metricsHistory: [],
        tileCounter: 0,
        totalTilesProduced: 0,
        totalTilesScrapped: 0,
        currentPeriodMetrics: {
          periodStartTick: 0,
          totalProduced: 0,
          firstQuality: 0,
          secondQuality: 0,
          thirdQuality: 0,
          scrap: 0,
          scrapByStation: {
            press: 0, dryer: 0, glaze: 0, printer: 0, kiln: 0, sorting: 0, packaging: 0,

        },
        machineStates: {

    },
    
    loadSimulation: async (sessionCode) => {

      console.log(`Loading simulation: ${sessionCode}`);
    },
    
    startSimulation: () => {
      set({ isRunning: true });
    },
    
    pauseSimulation: () => {
      set({ isRunning: false });
    },
    
    stopSimulation: async () => {
      set({ isRunning: false });
      // Trigger final sync
    },
    
    resetSimulation: () => {
      set({
        isRunning: false,
        currentSimTick: 0,
        currentProductionTick: 0,
      });
    },
    
    // -------------------------------------------------------------------------
    // MAIN TICK
    // -------------------------------------------------------------------------
    tick: () => {
      const state = get();
      if (!state.isRunning || !state.session) return;
      
      const newSimTick = state.currentSimTick + 1;
      const newProductionTick = Math.floor(newSimTick / state.config.productionTickRatio);
      
      // 1. Record machine states for all stations
      const machineStateIds: Record<StationName, string> = {} as any;
      for (const station of STATION_ORDER) {
        machineStateIds[station] = state.recordMachineState(station, newSimTick, newProductionTick);
      }
      
      // 2. Create new tile at press (every production tick)
      if (newSimTick % state.config.productionTickRatio === 0) {
        const tile = state.createTile(newSimTick, newProductionTick);
        
        // Record initial snapshot at press
        state.recordTileSnapshot(
          tile.id,
          'press',
          newSimTick,
          newProductionTick,
          machineStateIds.press
        );
      }
      
      // 3. Move tiles on conveyor
      state.moveTilesOnConveyor(newSimTick, newProductionTick);
      
      // 4. Random parameter changes
      if (Math.random() < state.config.parameterChangeChance) {
        const stations = Object.keys(state.currentParams) as StationName[];
        const randomStation = stations[Math.floor(Math.random() * stations.length)];
        // Apply random drift to a parameter
        // Implementation depends on station-specific logic
      }
      
      // 5. Check scenarios
      state.checkAndActivateScenarios(newSimTick);
      
      // 6. Update metrics
      state.updatePeriodMetrics(newSimTick, newProductionTick);
      
      // 7. Finalize period if needed
      if (newSimTick % state.config.metricsPeriodTicks === 0) {
        state.finalizePeriodMetrics(newSimTick, newProductionTick);
      }
      
      // Update tick counters
      set({
        currentSimTick: newSimTick,
        currentProductionTick: newProductionTick,
      });
    },
    
    // -------------------------------------------------------------------------
    // MACHINE STATE MANAGEMENT
    // -------------------------------------------------------------------------
    recordMachineState: (station, simTick, productionTick) => {
      const state = get();
      const id = nanoid();
      const params = state.currentParams[station];
      const status = state.machineStatus[station];
      
      const baseRecord = {
        id,
        simulation_id: state.session!.id,
        sim_tick: simTick,
        production_tick: productionTick,
        is_operating: status.isOperating,
        fault_code: status.faultCode,
        created_at: new Date().toISOString(),
        synced: false,
      };
      
      const record = { ...baseRecord, ...params };
      
      set((s) => {
        const newMap = new Map(s.machineStates[station]);
        newMap.set(simTick, record as any);
        return {
          machineStates: {
            ...s.machineStates,
            [station]: newMap,
          },
          unsyncedRecords: {
            ...s.unsyncedRecords,
            machineStates: [...s.unsyncedRecords.machineStates, { station, simTick }],
          },
        };
      });
      
      return id;
    },
    
    updateMachineParam: (station, param, value, changeType = 'drift') => {
      const state = get();
      const oldValue = (state.currentParams[station] as any)[param];
      
      // Record parameter change
      const changeRecord: ParameterChangeRecord = {
        id: nanoid(),
        simulation_id: state.session!.id,
        sim_tick: state.currentSimTick,
        production_tick: state.currentProductionTick,
        station,
        parameter_name: param,
        old_value: oldValue,
        new_value: value,
        change_magnitude: Math.abs(value - oldValue),
        change_pct: oldValue !== 0 ? ((value - oldValue) / oldValue) * 100 : undefined,
        change_type: changeType,
        synced: false,
      };
      
      set((s) => ({
        currentParams: {
          ...s.currentParams,
          [station]: {
            ...s.currentParams[station],
            [param]: value,
          },
        },
        parameterChanges: [...s.parameterChanges, changeRecord],
        unsyncedRecords: {
          ...s.unsyncedRecords,
          parameterChanges: [...s.unsyncedRecords.parameterChanges, changeRecord.id],
        },
      }));
    },
    
    setMachineFault: (station, faultCode) => {
      set((s) => ({
        machineStatus: {
          ...s.machineStatus,
          [station]: { isOperating: false, faultCode },
        },
      }));
    },
    
    clearMachineFault: (station) => {
      set((s) => ({
        machineStatus: {
          ...s.machineStatus,
          [station]: { isOperating: true, faultCode: undefined },
        },
      }));
    },
    
    // -------------------------------------------------------------------------
    // TILE MANAGEMENT
    // -------------------------------------------------------------------------
    createTile: (simTick, productionTick) => {
      const state = get();
      const tileNumber = state.tileCounter + 1;
      const id = nanoid();
      
      const tile: TileRecord = {
        id,
        simulation_id: state.session!.id,
        tile_number: tileNumber,
        created_at_sim_tick: simTick,
        created_at_production_tick: productionTick,
        status: 'in_production',
        current_station: 'press',
        final_grade: 'pending',
        synced: false,
      };
      
      // Add to conveyor
      const conveyorPosition: ConveyorPosition = {
        tile_id: id,
        current_station: 'press',
        position_in_station: 0,
        entered_at_sim_tick: simTick,
        next_station: 'dryer',
        ticks_until_next_station: state.config.stationGapProductionTicks * state.config.productionTickRatio,
      };
      
      set((s) => {
        const newTiles = new Map(s.tiles);
        newTiles.set(id, tile);
        
        const newTilesByNumber = new Map(s.tilesByNumber);
        newTilesByNumber.set(tileNumber, id);
        
        const newConveyor = new Map(s.conveyorPositions);
        newConveyor.set(id, conveyorPosition);
        
        return {
          tiles: newTiles,
          tilesByNumber: newTilesByNumber,
          conveyorPositions: newConveyor,
          tileCounter: tileNumber,
          unsyncedRecords: {
            ...s.unsyncedRecords,
            tiles: [...s.unsyncedRecords.tiles, id],
          },
        };
      });
      
      return tile;
    },
    
    updateTileStatus: (tileId, status) => {
      set((s) => {
        const newTiles = new Map(s.tiles);
        const tile = newTiles.get(tileId);
        if (tile) {
          newTiles.set(tileId, { ...tile, status, synced: false });
        }
        return { tiles: newTiles };
      });
    },
    
    setTileGrade: (tileId, grade) => {
      set((s) => {
        const newTiles = new Map(s.tiles);
        const tile = newTiles.get(tileId);
        if (tile) {
          newTiles.set(tileId, { ...tile, final_grade: grade, synced: false });
        }
        return { tiles: newTiles };
      });
    },
    
    scrapTile: (tileId, station, defects) => {
      set((s) => {
        const newTiles = new Map(s.tiles);
        const tile = newTiles.get(tileId);
        if (tile) {
          newTiles.set(tileId, {
            ...tile,
            status: `scrapped_at_${station}` as TileRecord['status'],
            final_grade: 'scrap',
            completed_at_sim_tick: s.currentSimTick,
            synced: false,
          });
        }
        
        // Remove from conveyor
        const newConveyor = new Map(s.conveyorPositions);
        newConveyor.delete(tileId);
        
        // Update metrics
        const newMetrics = { ...s.currentPeriodMetrics };
        newMetrics.scrap++;
        newMetrics.scrapByStation[station]++;
        defects.forEach((d) => {
          newMetrics.defectCounts[d] = (newMetrics.defectCounts[d] || 0) + 1;
        });
        
        return {
          tiles: newTiles,
          conveyorPositions: newConveyor,
          totalTilesScrapped: s.totalTilesScrapped + 1,
          currentPeriodMetrics: newMetrics,
        };
      });
    },
    
    // -------------------------------------------------------------------------
    // SNAPSHOT MANAGEMENT
    // -------------------------------------------------------------------------
    recordTileSnapshot: (tileId, station, simTick, productionTick, machineStateId, defectInfo) => {
      const state = get();
      const id = nanoid();
      
      const snapshot: TileSnapshotRecord = {
        id,
        tile_id: tileId,
        simulation_id: state.session!.id,
        station,
        station_order: STATION_ORDER_MAP[station],
        entry_sim_tick: simTick,
        entry_production_tick: productionTick,
        machine_state_id: machineStateId,
        parameters_snapshot: { ...state.currentParams[station] } as Record<string, unknown>,
        defect_detected: defectInfo?.detected ?? false,
        defect_types: defectInfo?.types,
        defect_severity: defectInfo?.severity,
        scrapped_here: defectInfo?.scrapped ?? false,
        synced: false,
      };
      
      set((s) => {
        const newSnapshots = new Map(s.tileSnapshots);
        const existing = newSnapshots.get(tileId) || [];
        newSnapshots.set(tileId, [...existing, snapshot]);
        
        return {
          tileSnapshots: newSnapshots,
          unsyncedRecords: {
            ...s.unsyncedRecords,
            snapshots: [...s.unsyncedRecords.snapshots, id],
          },
        };
      });
      
      return id;
    },
    
    // -------------------------------------------------------------------------
    // CONVEYOR MANAGEMENT
    // -------------------------------------------------------------------------
    moveTilesOnConveyor: (simTick, productionTick) => {
      const state = get();
      
      set((s) => {
        const newConveyor = new Map(s.conveyorPositions);
        const newTiles = new Map(s.tiles);
        let newTotalProduced = s.totalTilesProduced;
        const newMetrics = { ...s.currentPeriodMetrics };
        
        for (const [tileId, position] of newConveyor) {
          position.ticks_until_next_station--;
          
          if (position.ticks_until_next_station <= 0) {
            const currentIndex = STATION_ORDER.indexOf(position.current_station as StationName);
            
            if (currentIndex < STATION_ORDER.length - 1) {
              // Move to next station
              const nextStation = STATION_ORDER[currentIndex + 1];
              
              // Get machine state ID for this station
              const machineStateRecord = s.machineStates[nextStation].get(simTick);
              const machineStateId = machineStateRecord?.id || '';
              
              // Record snapshot
              get().recordTileSnapshot(tileId, nextStation, simTick, productionTick, machineStateId);
              
              // Update position
              position.current_station = nextStation;
              position.next_station = STATION_ORDER[currentIndex + 2];
              position.ticks_until_next_station = s.config.stationGapProductionTicks * s.config.productionTickRatio;
              position.entered_at_sim_tick = simTick;
              
              // Update tile
              const tile = newTiles.get(tileId);
              if (tile) {
                newTiles.set(tileId, { ...tile, current_station: nextStation, synced: false });
              }
            } else {
              // Tile completed - determine grade based on sorting
              const tile = newTiles.get(tileId);
              if (tile) {
                // Simple grading logic - can be expanded
                const snapshots = s.tileSnapshots.get(tileId) || [];
                const hasDefects = snapshots.some((snap) => snap.defect_detected);
                const grade: QualityGrade = hasDefects ? 'second_quality' : 'first_quality';
                
                newTiles.set(tileId, {
                  ...tile,
                  status: 'completed',
                  final_grade: grade,
                  completed_at_sim_tick: simTick,
                  synced: false,
                });
                
                // Update metrics
                newTotalProduced++;
                newMetrics.totalProduced++;
                if (grade === 'first_quality') newMetrics.firstQuality++;
                else if (grade === 'second_quality') newMetrics.secondQuality++;
                else if (grade === 'third_quality') newMetrics.thirdQuality++;
              }
              
              // Remove from conveyor
              newConveyor.delete(tileId);
            }
          }
        }
        
        return {
          conveyorPositions: newConveyor,
          tiles: newTiles,
          totalTilesProduced: newTotalProduced,
          currentPeriodMetrics: newMetrics,
        };
      });
    },
    
    // -------------------------------------------------------------------------
    // SCENARIO MANAGEMENT
    // -------------------------------------------------------------------------
    checkAndActivateScenarios: (simTick) => {
      // Check trigger conditions against current parameters
      // Activate matching scenarios
    },
    
    deactivateScenario: (scenarioId, simTick) => {
      set((s) => {
        const newActive = new Map(s.activeScenarios);
        const scenario = newActive.get(scenarioId);
        if (scenario) {
          scenario.deactivated_at_sim_tick = simTick;
          scenario.is_active = false;
          newActive.delete(scenarioId);
          return {
            activeScenarios: newActive,
            scenarioHistory: [...s.scenarioHistory, scenario],
          };
        }
        return {};
      });
    },
    
    // -------------------------------------------------------------------------
    // METRICS
    // -------------------------------------------------------------------------
    updatePeriodMetrics: (simTick, productionTick) => {
      // Updates happen in other methods (createTile, scrapTile, etc.)
    },
    
    finalizePeriodMetrics: (simTick, productionTick) => {
      const state = get();
      const metrics = state.currentPeriodMetrics;
      
      // Calculate OEE components
      const quality_pct = metrics.totalProduced > 0
        ? (metrics.firstQuality / metrics.totalProduced) * 100
        : 100;
      
      const record: ProductionMetricsRecord = {
        id: nanoid(),
        simulation_id: state.session!.id,
        period_start_sim_tick: metrics.periodStartTick,
        period_end_sim_tick: simTick,
        period_start_production_tick: Math.floor(metrics.periodStartTick / state.config.productionTickRatio),
        period_end_production_tick: productionTick,
        total_tiles_produced: metrics.totalProduced,
        first_quality_count: metrics.firstQuality,
        second_quality_count: metrics.secondQuality,
        third_quality_count: metrics.thirdQuality,
        scrap_count: metrics.scrap,
        quality_pct,
        scrap_by_station: metrics.scrapByStation,
        defect_counts: metrics.defectCounts,
        synced: false,
      };
      
      set((s) => ({
        metricsHistory: [...s.metricsHistory, record],
        currentPeriodMetrics: {
          periodStartTick: simTick,
          totalProduced: 0,
          firstQuality: 0,
          secondQuality: 0,
          thirdQuality: 0,
          scrap: 0,
          scrapByStation: {
            press: 0, dryer: 0, glaze: 0, printer: 0, kiln: 0, sorting: 0, packaging: 0,
          },
          defectCounts: {},
        },
        unsyncedRecords: {
          ...s.unsyncedRecords,
          metrics: [...s.unsyncedRecords.metrics, record.id],
        },
      }));
    },
    
    // -------------------------------------------------------------------------
    // SYNC
    // -------------------------------------------------------------------------
    markForSync: (type, id) => {
      set((s) => ({
        unsyncedRecords: {
          ...s.unsyncedRecords,
          [type]: [...s.unsyncedRecords[type], id],
        },
      }));
    },
    
    getUnsyncedData: () => {
      const state = get();
      
      // Collect unsynced machine states
      const machineStates: Record<StationName, any[]> = {
        press: [], dryer: [], glaze: [], printer: [], kiln: [], sorting: [], packaging: [],
      };
      
      for (const { station, simTick } of state.unsyncedRecords.machineStates) {
        const record = state.machineStates[station].get(simTick);
        if (record && !record.synced) {
          machineStates[station].push(record);
        }
      }
      
      // Collect unsynced tiles
      const tiles = state.unsyncedRecords.tiles
        .map((id) => state.tiles.get(id))
        .filter((t): t is TileRecord => t !== undefined && !t.synced);
      
      // Collect unsynced snapshots
      const snapshots: TileSnapshotRecord[] = [];
      for (const id of state.unsyncedRecords.snapshots) {
        for (const [, snaps] of state.tileSnapshots) {
          const snap = snaps.find((s) => s.id === id);
          if (snap && !snap.synced) {
            snapshots.push(snap);
          }
        }
      }
      
      // Collect unsynced parameter changes
      const parameterChanges = state.parameterChanges.filter(
        (p) => state.unsyncedRecords.parameterChanges.includes(p.id) && !p.synced
      );
      
      // Collect unsynced scenarios
      const scenarios = state.scenarioHistory.filter(
        (s) => state.unsyncedRecords.scenarios.includes(s.id) && !s.synced
      );
      
      // Collect unsynced metrics
      const metrics = state.metricsHistory.filter(
        (m) => state.unsyncedRecords.metrics.includes(m.id) && !m.synced
      );
      
      return { machineStates, tiles, snapshots, parameterChanges, scenarios, metrics };
    },
    
    markAsSynced: (type, ids) => {
      set((s) => ({
        unsyncedRecords: {
          ...s.unsyncedRecords,
          [type]: s.unsyncedRecords[type].filter((id: any) => 
            typeof id === 'string' ? !ids.includes(id) : true
          ),
        },
      }));
    },
    
    // -------------------------------------------------------------------------
    // QUERIES
    // -------------------------------------------------------------------------
    getTileById: (id) => get().tiles.get(id),
    
    getTileByNumber: (number) => {
      const id = get().tilesByNumber.get(number);
      return id ? get().tiles.get(id) : undefined;
    },
    
    getTileSnapshots: (tileId) => get().tileSnapshots.get(tileId) || [],
    
    getMachineStateAtTick: (station, simTick) => get().machineStates[station].get(simTick),
    
    getRecentParameterChanges: (station, limit = 10) => {
      const changes = get().parameterChanges;
      const filtered = station
        ? changes.filter((c) => c.station === station)
        : changes;
      return filtered.slice(-limit);
    },
  }))
);

#### 6.3: Create Sync Service

Create `src/services/syncService.ts`:

```typescript
// =============================================================================
// SYNC SERVICE - Batch sync Zustand to Supabase
// =============================================================================

import { supabase } from '@/lib/supabase';
import { useSimulationStore } from '@/store/simulationStore';
import type { StationName } from '@/types/simulator';

const MACHINE_TABLES: Record<StationName, string> = {
  press: 'machine_press_states',
  dryer: 'machine_dryer_states',
  glaze: 'machine_glaze_states',
  printer: 'machine_printer_states',
  kiln: 'machine_kiln_states',
  sorting: 'machine_sorting_states',
  packaging: 'machine_packaging_states',
};

class SyncService {
  private syncInterval: NodeJS.Timeout | null = null;
  private isSyncing = false;
  private readonly SYNC_INTERVAL_MS = 2000;

  start() {
    if (this.syncInterval) return;
    
    this.syncInterval = setInterval(() => {
      this.sync();
    }, this.SYNC_INTERVAL_MS);
    
    console.log('Sync service started');
  }

  stop() {
    if (this.syncInterval) {
      clearInterval(this.syncInterval);
      this.syncInterval = null;
    }
    
    // Final sync
    return this.sync();
  }

  async sync() {
    if (this.isSyncing) return;
    this.isSyncing = true;

    try {
      const store = useSimulationStore.getState();
      const unsynced = store.getUnsyncedData();
      
      const promises: Promise<void>[] = [];

      // Sync machine states
      for (const [station, records] of Object.entries(unsynced.machineStates)) {
        if (records.length > 0) {
          const tableName = MACHINE_TABLES[station as StationName];
          // Remove local-only fields before insert
          const cleanRecords = records.map(({ synced, ...rest }) => rest);
          
          promises.push(
            supabase
              .from(tableName)
              .insert(cleanRecords)
              .then(({ error }) => {
                if (error) throw error;
                // Mark as synced
                store.markAsSynced('machineStates', records.map((r: any) => r.id));
              })
          );
        }
      }

      // Sync tiles
      if (unsynced.tiles.length > 0) {
        const cleanTiles = unsynced.tiles.map(({ synced, ...rest }) => rest);
        promises.push(
          supabase
            .from('tiles')
            .upsert(cleanTiles)
            .then(({ error }) => {
              if (error) throw error;
              store.markAsSynced('tiles', unsynced.tiles.map((t) => t.id));
            })
        );
      }

      // Sync snapshots
      if (unsynced.snapshots.length > 0) {
        const cleanSnapshots = unsynced.snapshots.map(({ synced, ...rest }) => rest);
        promises.push(
          supabase
            .from('tile_station_snapshots')
            .insert(cleanSnapshots)
            .then(({ error }) => {
              if (error) throw error;
              store.markAsSynced('snapshots', unsynced.snapshots.map((s) => s.id));
            })
        );
      }

      // Sync parameter changes
      if (unsynced.parameterChanges.length > 0) {
        const cleanChanges = unsynced.parameterChanges.map(({ synced, ...rest }) => rest);
        promises.push(
          supabase
            .from('parameter_change_events')
            .insert(cleanChanges)
            .then(({ error }) => {
              if (error) throw error;
              store.markAsSynced('parameterChanges', unsynced.parameterChanges.map((c) => c.id));
            })
        );
      }

      // Sync metrics
      if (unsynced.metrics.length > 0) {
        const cleanMetrics = unsynced.metrics.map(({ synced, ...rest }) => rest);
        promises.push(
          supabase
            .from('production_metrics')
            .insert(cleanMetrics)
            .then(({ error }) => {
              if (error) throw error;
              store.markAsSynced('metrics', unsynced.metrics.map((m) => m.id));
            })
        );
      }

      await Promise.all(promises);
      
      const totalSynced = 
        Object.values(unsynced.machineStates).reduce((sum, arr) => sum + arr.length, 0) +
        unsynced.tiles.length +
        unsynced.snapshots.length +
        unsynced.parameterChanges.length +
        unsynced.metrics.length;
      
      if (totalSynced > 0) {
        console.log(`Synced ${totalSynced} records to Supabase`);
      }
    } catch (error) {
      console.error('Sync error:', error);
    } finally {
      this.isSyncing = false;
    }
  }
}

export const syncService = new SyncService();
```

import { useEffect, useRef } from 'react';
import { useSimulationStore } from '@/store/simulationStore';
import { syncService } from '@/services/syncService';

export function SimulationRunner() {
  const { 
    isRunning, 
    config,
    tick,
    sessionCode,
  } = useSimulationStore();
  
  const intervalRef = useRef<NodeJS.Timeout | null>(null);

  // Start sync service when simulation starts
  useEffect(() => {
    if (isRunning && sessionCode) {
      syncService.start();
      console.log(`Simulation ${sessionCode} running - sync service started`);
    }
    
    return () => {
      if (!isRunning) {
        syncService.stop().then(() => {
          console.log('Final sync completed');
        });
      }
    };
  }, [isRunning, sessionCode]);

  // Main tick loop
  useEffect(() => {
    if (isRunning) {
      intervalRef.current = setInterval(() => {
        tick();
      }, config.tickDurationMs);
    } else {
      if (intervalRef.current) {
        clearInterval(intervalRef.current);
        intervalRef.current = null;
      }
    }

    return () => {
      if (intervalRef.current) {
        clearInterval(intervalRef.current);
      }
    };
  }, [isRunning, config.tickDurationMs, tick]);

  return null; // This is a logic-only component
}

import type { StationName } from '@/types/simulator';
import {

  DRYER_OPTIMAL_RANGES,

  PRINTER_OPTIMAL_RANGES,
  KILN_OPTIMAL_RANGES,
} from '@/types/simulator';

const OPTIMAL_RANGES: Record<string, any> = {
  press: PRESS_OPTIMAL_RANGES,
  dryer: DRYER_OPTIMAL_RANGES,
  glaze: GLAZE_OPTIMAL_RANGES,
  printer: PRINTER_OPTIMAL_RANGES,
  kiln: KILN_OPTIMAL_RANGES,
  sorting: {},
  packaging: {},
};

// Hook for individual machine state
export function useMachineState(station: StationName) {
  const currentParams = useSimulationStore((s) => s.currentParams[station]);
  const machineStatus = useSimulationStore((s) => s.machineStatus[station]);
  const recentChanges = useSimulationStore((s) => 
    s.getRecentParameterChanges(station, 5)
  );
  
  // Calculate deviations from optimal
  const ranges = OPTIMAL_RANGES[station];
  const deviations: Record<string, { value: number; status: 'ok' | 'warning' | 'critical' }> = {};
  
  for (const [param, range] of Object.entries(ranges)) {
    const value = (currentParams as any)[param];
    if (value !== undefined && range && 'min' in (range as any) && 'max' in (range as any)) {
      const r = range as { min: number; max: number };
      const mid = (r.min + r.max) / 2;
      const tolerance = (r.max - r.min) / 2;
      const deviation = Math.abs(value - mid) / tolerance;
      
      deviations[param] = {
        value: deviation,
        status: deviation > 1.5 ? 'critical' : deviation > 1 ? 'warning' : 'ok',
      };
    }
  }
  
  return {
    params: currentParams,
    status: machineStatus,
    recentChanges,
    deviations,
    isHealthy: Object.values(deviations).every((d) => d.status === 'ok'),
  };
}
